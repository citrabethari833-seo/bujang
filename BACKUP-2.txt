COMMAND MENYEBAR  htaccess KE SELURUH DALAM FOLDER 
find /path/to/your/ -type d -exec cp .htaccess {} \;

COMMAND MENYALIN FILE htaccess HANYA KE FOLDER YANG BELUM MEMILIKINYA
find /path/to/your/ -type d ! -exec test -e "{}/.htaccess" \; -exec cp /path/to/your/.htaccess "{}/.htaccess" \;

COMMAND MENGGANTI PERMISSIONS FILE KE SELURUH DALAM FOLDER 
find /path/folder -type f -name "nama_file" -exec chmod 644 {} \;

COMMAND MENGHAPUS FILE KE SELURUH DALAM FOLDER 
find /path/folder -type f -name "nama_file" -delete
